CSS Classes Reference
/* Containers */
.ux-container       /* Max-width 1280px, centered */
.ux-container-fluid /* Full width */

/* Row */
.ux-row             /* 12-column grid container */

/* Base Columns (mobile-first) */
.ux-col-1 to .ux-col-12    /* Default (all screens) */

/* Responsive Breakpoints */
.ux-col-md-1 to .ux-col-md-12  /* Medium (768px+) */
.ux-col-lg-1 to .ux-col-lg-12  /* Large (1024px+) */
.ux-col-xl-1 to .ux-col-xl-12  /* XL (1280px+) */

/* Breakpoint Values */
md:  768px   /* Tablets */
lg:  1024px  /* Desktop */
xl:  1280px  /* Large desktop */

  <div class="ux-tabs-panels">

  </div>

HTML - AJAX
<div class="ux-tabs">
  <div class="ux-tabs-list">
    <button class="ux-tab is-active" data-url="/api/tab1.json">Tab 1</button>
    <button class="ux-tab" data-url="/api/tab2.json">Tab 2</button>
    <button class="ux-tab" data-url="/api/tab3.json" data-cache="false">No Cache</button>
  </div>
  <div class="ux-tabs-panels">
    <div class="ux-tab-panel is-active"></div>
    <div class="ux-tab-panel"></div>
    <div class="ux-tab-panel"></div>
  </div>
</div>
Data Attributes & API
/* Data Attributes (on .ux-tab) */
data-url="..."         /* URL to fetch content from */
data-cache="false"    /* Disable caching, refetch on click */

/* Events */
ux-tabs:change   /* Fired when tab changes { index, tab, panel } */
ux-tabs:loaded   /* Fired when AJAX content loads { index, content } */
ux-tabs:error    /* Fired on AJAX error { index, error } */

  </div>

CSS Classes Reference
/* Base Classes */

.ux-alert-icon     /* Icon wrapper */
.ux-alert-content  /* Content wrapper */

/* Variant Classes (add to .ux-alert) */

.ux-alert-success  /* Green - success action */
.ux-alert-warning  /* Yellow - warning/caution */
.ux-alert-error    /* Red - error/danger */
